001package Job;
002/**
003 * Interface untuk menambahkan atau menghapus ke/dari daftar obeserver dan memberi notifikasi ke obeserver
004 */
005public interface Subject 
006{
007    /**
008     * method untuk menmbahkan objek ke dalam daftar observer
009     * @param o
010     */
011    public void attach(Observer o);
012
013    /**
014     * method untuk menghapus objek dari daftar observer
015     * @param o
016     */
017    public void detach(Observer o);
018
019    /**
020     * method untuk memberi notifikasi ke observer
021     * @param m
022     */
023    public void notifyUpdate(Message m);
024}
